scp svn.mpi.nl/LAT/PluginsAndModules/metadatasearch/trunk/yams-cs-connector/target/yams-cs-connector-1.0.14-pretesting-r40997-mpi-deploy.tar.gz petwit@lux16.mpi.nl:/lat/webapps/dists/
sudo su latuser
cd /lat/webapps/yams-cs-connector; tar -xzvf /lat/webapps/dists/yams-cs-connector-1.0.14-pretesting-r40997-mpi-deploy.tar.gz
cd /lat/webapps/yams-cs-connector/yams-cs-connector-1.0.14-pretesting-r40997/war; jar -xfv yams-cs-connector-1.0.14-pretesting.war
rm /lat/webapps/yams-cs-connector/current
